

``` C++

#include <iostream>
#include <string>

using namespace std;

// 로봇 클래스
class Robot {
private:
    string name;        // 로봇 이름
    int battery;        // 배터리 잔량 (0~100)
    int positionX;      // X 좌표
    int positionY;      // Y 좌표

public:
    // 생성자
    Robot(string n) : name(n), battery(100), positionX(0), positionY(0) {
        cout << "[" << name << "] 로봇이 생성되었습니다." << endl;
    }

    // 이동 기능
    void move(int dx, int dy) {
        if (battery >= 10) {
            battery -= 10;
            positionX += dx;
            positionY += dy;
            cout << "[" << name << "] 이동 완료! 위치: (" << positionX << ", " << positionY 
                 << "), 남은 배터리: " << battery << "%" << endl;
        } else {
            cout << "[" << name << "] 배터리 부족! 이동할 수 없습니다. (현재 배터리: " << battery << "%)" << endl;
        }
    }

    // 충전 기능
    void charge() {
        battery = 100;
        cout << "[" << name << "] 충전 완료! 배터리: 100%" << endl;
    }

    // 상태 확인
    void showStatus() {
        cout << "[" << name << "] 상태 - 위치: (" << positionX << ", " << positionY 
             << "), 배터리: " << battery << "%" << endl;
    }

    // 배터리 잔량 반환
    int getBattery() {
        return battery;
    }
};

int main() {
    // 로봇 생성
    Robot myRobot("Robo");

    cout << "\n=== 로봇 작동동동동동동 테스트 ===" << endl;
    
    // 초기 상태 확인
    myRobot.showStatus();
    cout << endl;

    // 이동 테스트
    cout << "--- 이동 테스트 ---" << endl;
    myRobot.move(5, 3);   // (5, 3)으로 이동
    myRobot.move(2, -1);  // (2, -1)만큼 추가 이동
    myRobot.move(-3, 4);  // (-3, 4)만큼 추가 이동
    cout << endl;

    // 상태 확인
    myRobot.showStatus();
    cout << endl;

    // 배터리 소모 테스트
    cout << "--- 배터리 소모 테스트 ---" << endl;
    for (int i = 0; i < 8; i++) {
        myRobot.move(1, 0);
    }
    cout << endl;

    // 충전 테스트
    cout << "--- 충전 테스트 ---" << endl;
    myRobot.charge();
    myRobot.showStatus();
    cout << endl;

    // 충전 후 이동
    cout << "--- 충전 후 이동 ---" << endl;
    myRobot.move(10, 10);

    return 0;
}
```
